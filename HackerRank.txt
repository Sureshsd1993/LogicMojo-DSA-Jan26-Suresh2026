class Result
{

    /*
     * Complete the 'sort_an_array' function below.
     *
     * The function is expected to return an INTEGER_ARRAY.
     * The function accepts following parameters:
     *  1. INTEGER n
     *  2. INTEGER_ARRAY arr
     */

    public static List<int> sort_an_array(int n, List<int> arr)
    {
       
       int i =0; int j =0; int k = n-1;
        
        while(j <= k){
            if(arr[j] == 1){
                j++;
            }
            else if(arr[j] == 2){
                int temp = arr[j];
                arr[j] = arr[k];
                arr[k] = temp;
                k--;
            }
            else{
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
                j++;
                i++;
            }
        }
        
       return arr;
    }

}

class Solution
{
    public static void Main(string[] args)
    {
         
        int n = 5;
        List<int> arr = new List<int> {0,1,2,1,2};
        List<int> arrList = Result.sort_an_array(n,arr);
        Console.WriteLine("Answer : {0}", string.Join(",",arrList));
    }
}

     *
     * The function is expected to return an INTEGER_ARRAY.
     * The function accepts following parameters:
     *  1. INTEGER n
     *  2. INTEGER_ARRAY arr
     */

    public static List<int> move_func(int n, List<int> arr)
    {
        
        int i =0; int j =0; 
        
        while(j < n){
            if(arr[j] != 0){
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
                i++;
                j++;
            }else{
                j++;
            }
        }

        return arr;
    }

}

     */

    public static int max_consecutive(List<int> arr)
    {
        int count = 0;
        int maxCount = 0;
        for(int i=0; i< arr.Count; i++){
            if(arr[i] == 1){
                count++;
                maxCount = Math.Max(maxCount, count);
            }
            else{
                count =0;
            }
        }
        
        return maxCount;
    }

}

class Result
{

    /*
     * Complete the 'maxProfit' function below.
     *
     * The function is expected to return an INTEGER.
     * The function accepts following parameters:
     *  1. INTEGER_ARRAY nums
     *  2. INTEGER n
     */

    public static int maxProfit(List<int> nums, int n)
    {

        int maxProfit = 0;
        for(int i=1; i< nums.Count; i++){
            
           if(nums[i] > nums[i-1]){
               maxProfit += nums[i] - nums[i-1];
           }
        }
        
        return maxProfit;
    }

}

Find Missing & Repeating Number

using System.Collections.Generic;
using System.Collections;
using System.ComponentModel;
using System.Diagnostics.CodeAnalysis;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Reflection;
using System.Runtime.Serialization;
using System.Text.RegularExpressions;
using System.Text;

class Result
{

    /*
     * Complete the 'find_missing' function below.
     *
     * The function is expected to return an INTEGER_ARRAY.
     * The function accepts INTEGER_ARRAY arr as parameter.
     */

    public static List<int> find_missing(List<int> arr)
    {
        int repeat = 0;
        int missing = 0;
        bool[] visted = new bool [arr.Count +1];
        List<int> result = new List<int>();
        
        foreach(int num in arr){
            if(visted[num]){
                repeat = num;
            }else{
                visted[num] = true;
            }
        }
        
        for(int i=1 ;i <= arr.Count; i++){
            if(!visted[i]){
                missing = i;
            }
        }
        result.Add(repeat);
        result.Add(missing);
        return result;
        
    }

}

class Solution
{
    public static void Main(string[] args)
    {
        TextWriter textWriter = new StreamWriter(@System.Environment.GetEnvironmentVariable("OUTPUT_PATH"), true);

        int n = Convert.ToInt32(Console.ReadLine().Trim());

        List<int> arr = Console.ReadLine().TrimEnd().Split(' ').ToList().Select(arrTemp => Convert.ToInt32(arrTemp)).ToList();

        List<int> result = Result.find_missing(arr);

        textWriter.WriteLine(String.Join(" ", result));

        textWriter.Flush();
        textWriter.Close();
    }
}


//Majority Element (Boyer-Moore Voting Algorithm)

using System.CodeDom.Compiler;
using System.Collections.Generic;
using System.Collections;
using System.ComponentModel;
using System.Diagnostics.CodeAnalysis;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Reflection;
using System.Runtime.Serialization;
using System.Text.RegularExpressions;
using System.Text;
using System;

class Result
{

    /*
     * Complete the 'majorityElement' function below.
     *
     * The function is expected to return an INTEGER.
     * The function accepts following parameters:
     *  1. INTEGER n
     *  2. INTEGER_ARRAY arr
     */

    public static int majorityElement(int n, List<int> arr)
    {
                int candidate = 0;
                int count = 0;
                int j = 0;
                while(j < arr.Count){

                    if(count == 0){
                        candidate = arr[j];
                        count = 1;
                    }
                    else if(candidate == arr[j]){
                        count++;
                    }
                    else{
                        count--;
                    }
                    j++;
                }

                count = 0;
                foreach(var num in arr)
                {
                    if(num == candidate)
                    {
                        count++;
                    }
                }
        
                return (count > arr.Count / 2) ? candidate : -1;
            }

}

{
    public static void Main(string[] args)
    {
        TextWriter textWriter = new StreamWriter(@System.Environment.GetEnvironmentVariable("OUTPUT_PATH"), true);

        int n = Convert.ToInt32(Console.ReadLine().Trim());

        List<int> arr = Console.ReadLine().TrimEnd().Split(' ').ToList().Select(arrTemp => Convert.ToInt32(arrTemp)).ToList();

        int result = Result.majorityElement(n, arr);

        textWriter.WriteLine(result);

        textWriter.Flush();
        textWriter.Close();
    }
}

//Trapping rain water problem , Time Complexity - O(n), Space Complexity O(1)

using System.CodeDom.Compiler;
using System.Collections.Generic;
using System.Collections;
using System.ComponentModel;
using System.Diagnostics.CodeAnalysis;
using System.Globalization;
using System.IO;
using System.Linq;
using System.Reflection;
using System.Runtime.Serialization;
using System.Text.RegularExpressions;
using System.Text;
using System;

class Result
{

    /*
     * Complete the 'rain_water' function below.
     *
     * The function is expected to return an INTEGER.
     * The function accepts INTEGER_ARRAY hei as parameter.
     */

    public static int rain_water(List<int> hei)
    {
        
        if(hei == null || hei.Count == 0){
            return 0;
        }
        
        int left=0;
        int right = hei.Count -1;
        int leftmax = 0;
        int rightmax = 0;
        int water =0;
        
        while(left < right){
            
            if(hei[left] < hei[right]){
                
                if(hei[left] >= leftmax){
                    leftmax = hei[left];
                }
                else{
                    water += leftmax - hei[left];
                }
                left++;
            }
            else{
    
                if(hei[right] >= rightmax){
                    rightmax = hei[right];
                }
                else{
                    water += rightmax - hei[right];
                }
                right--;
            
            } 
        }
        return water;

    }

}

class Solution
{
    public static void Main(string[] args)
    {
        TextWriter textWriter = new StreamWriter(@System.Environment.GetEnvironmentVariable("OUTPUT_PATH"), true);

        int n = Convert.ToInt32(Console.ReadLine().Trim());

        List<int> hei = Console.ReadLine().TrimEnd().Split(' ').ToList().Select(heiTemp => Convert.ToInt32(heiTemp)).ToList();

        int result = Result.rain_water(hei);

        textWriter.WriteLine(result);

        textWriter.Flush();
        textWriter.Close();
    }
}
